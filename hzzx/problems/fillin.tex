% 12.
\begin{question}
  已知 $\vec{a}=(2,3)$, $\vec{b}=(-1,0)$, 则 $\vec{a}$ 在 $\vec{b}$ 方向上的投影向量为 \fillin[$(2, 0)$]（用坐标表示）.
  \begin{solution}
    \textbf{以下均为占位符}
    由 $f(x) = f(-x)$，得 $a = 1$。
  \end{solution}
\end{question}

\end{question}
